-- 1. Tabla de préstamos
CREATE TABLE IF NOT EXISTS public.loans (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    amount NUMERIC NOT NULL,
    payment_term TEXT NOT NULL,
    payment_method TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'pending' 
        CHECK (status IN ('pending', 'accepted', 'rejected', 'paid', 'overdue')),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT now() NOT NULL,
    due_date TIMESTAMP WITH TIME ZONE,
    interest_accrued NUMERIC DEFAULT 0 NOT NULL
);

-- 2. Seguridad RLS
ALTER TABLE public.loans ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view their own loans" ON public.loans FOR SELECT TO authenticated USING (auth.uid() = user_id);
CREATE POLICY "Users can insert their own loans" ON public.loans FOR INSERT TO authenticated WITH CHECK (auth.uid() = user_id);

-- 3. Realtime
ALTER PUBLICATION supabase_realtime ADD TABLE public.loans;

-- 1. Crear tabla de perfiles para manejar los roles
create table public.profiles (
  id uuid references auth.users on delete cascade primary key,
  email text,
  role text check (role in ('admin', 'cobrador')),
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 2. Habilitar seguridad de nivel de fila (RLS)
alter table public.profiles enable row level security;

-- 3. Políticas de acceso
create policy "Los perfiles son visibles por sus dueños." on public.profiles
  for select using (auth.uid() = id);

create policy "Los admins pueden ver todos los perfiles." on public.profiles
  for select using (
    exists (select 1 from public.profiles where id = auth.uid() and role = 'admin')
  );



-- 1. Eliminamos la política que causa el bucle infinito
DROP POLICY IF EXISTS "Los admins pueden ver todos los perfiles." ON public.profiles;

-- 2. Creamos una política simple para profiles (SIN RECURSIÓN)
-- Cada usuario solo necesita ver su propio perfil para que el check de préstamos funcione.
DROP POLICY IF EXISTS "Usuarios ven su propio perfil" ON public.profiles;
CREATE POLICY "Usuarios ven su propio perfil" 
ON public.profiles 
FOR SELECT 
TO authenticated 
USING (auth.uid() = id);

-- 3. La política de préstamos (loans) se mantiene así (esta no causa recursión)
DROP POLICY IF EXISTS "Admins and collectors can view all loans" ON public.loans;
CREATE POLICY "Admins and collectors can view all loans" 
ON public.loans 
FOR SELECT 
TO authenticated 
USING (
  exists (
    select 1 from public.profiles 
    where profiles.id = auth.uid() 
    and (profiles.role = 'admin' or profiles.role = 'cobrador')
  )
);

CREATE OR REPLACE FUNCTION get_loan_details(p_loan_id UUID)
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_loan             RECORD;
  v_amount_received  NUMERIC;
  v_total_interest   NUMERIC;
  v_initial_total    NUMERIC;
  v_disbursement     TIMESTAMPTZ;
  v_expiration       TIMESTAMPTZ;
  v_delay_days       INTEGER;
  v_late_interest    NUMERIC;
  v_total_to_pay     NUMERIC;
BEGIN

  SELECT *
  INTO v_loan
  FROM loans
  WHERE id = p_loan_id
    AND user_id = auth.uid();

  IF NOT FOUND THEN
    RAISE EXCEPTION 'Préstamo no encontrado o sin acceso';
  END IF;

  v_amount_received := v_loan.amount * 0.6;
  v_total_interest  := v_loan.amount * 0.4;
  v_initial_total   := v_amount_received + v_total_interest;

  -- ✅ Fechas sin conversión de zona horaria
  v_disbursement := COALESCE(v_loan.disbursed_at, v_loan.created_at);
  v_expiration   := v_disbursement + (v_loan.payment_term || ' days')::INTERVAL;

  v_delay_days    := 0;
  v_late_interest := 0;

  -- ✅ Solo NOW() usa zona horaria México para calcular mora
  IF (NOW() AT TIME ZONE 'America/Mexico_City') > v_expiration THEN
    v_delay_days    := EXTRACT(DAY FROM (NOW() AT TIME ZONE 'America/Mexico_City') - v_expiration)::INTEGER;
    v_late_interest := v_initial_total * 0.05 * v_delay_days;
  END IF;

  v_total_to_pay := v_initial_total + v_late_interest;

  RETURN json_build_object(
    'loan_id',          v_loan.id,
    'status',           v_loan.status,
    'requested_amount', v_loan.amount,
    'amount_received',  ROUND(v_amount_received, 2),
    'total_interest',   ROUND(v_total_interest, 2),
    'late_interest',    ROUND(v_late_interest, 2),
    'delay_days',       v_delay_days,
    'total_to_pay',     ROUND(v_total_to_pay, 2),
    'payment_term',     v_loan.payment_term,
    'payment_method',   v_loan.payment_method,
    'disbursement_date', TO_CHAR(v_disbursement, 'YYYY-MM-DD'),
    'expiration_date',   TO_CHAR(v_expiration,   'YYYY-MM-DD'),
    'is_overdue',       (NOW() AT TIME ZONE 'America/Mexico_City') > v_expiration
  );

END;
$$;

GRANT EXECUTE ON FUNCTION get_loan_details(UUID) TO authenticated;
